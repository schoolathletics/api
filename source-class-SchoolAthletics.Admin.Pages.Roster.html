123     public static function getMembers($sport,$season,$import = null){
124         //Accept sport as an object or ID
125         if(is_object($sport)){
126             $sport = $sport->term_id;
127         }
128         //Accept season as an object or ID
129         if(is_object($season)){
130             $season = $season->term_id;
131         }
132 
133         $_members = \SchoolAthletics\Roster::getMembers($sport,$season);
134 
135         $members = array();
136         foreach ($_members as $member) {
137             $status = get_the_terms($member,'sa_athlete_status');
138             $status = (is_array($status)) ? array_pop($status) : null;
139             $status = ($status) ? $status->name : '- - -';
140             $members[] = array(
141                     'ID'     => $member->ID,
142                     'photo'  => get_post_thumbnail_id( $member->ID ),
143                     'jersey' => get_post_meta( $member->ID, 'sa_jersey', true ),
144                     'name'   => get_post_meta( $member->ID, 'sa_name', true ),
145                     'height' => get_post_meta( $member->ID, 'sa_height', true ),
146                     'weight' => get_post_meta( $member->ID, 'sa_weight', true ),
147                     'order'  => get_post_meta( $member->ID, 'sa_order', true ),
148                     'status' => $status,
149                 );
150         }
151         //Defaults for athletes
152         $defaults = self::athleteDefaults();
153         if(!empty($import) && is_array($import)){
154             foreach ($import as $key => $value) {
155                 //$import[$key]['ID'] = '';
156                 $import[$key] = wp_parse_args($import[$key],$defaults);
157             }
158             $members = array_merge($members,$import);
159         }
160         if(empty($members)){
161             //Adds a new row to the bottom
162             $members[] = $defaults;
163         }
164         return $members;
165     }

169     /**
170      * Save roster
171      */
172     public static function save(){
173         if ( empty( $_REQUEST['_wpnonce'] ) || ! wp_verify_nonce( $_REQUEST['_wpnonce'], 'schoolathletics-save-roster' ) ) {
174             \SchoolAthletics\Admin\Notice::error(__( 'Failed to save roster. Please refresh the page and retry.', 'school-athletics' ));
175             return false;
176         }
177 
178         if(!empty($_REQUEST['season']) && !empty($_REQUEST['sport'])){
179             $roster_id = self::save_roster($_REQUEST['sport'],$_REQUEST['season'],$_POST);
180             $athletes = $_POST['athlete'];
181             $athlete_terms = array();
182             foreach ($athletes as $athlete) {
183                 $athlete_id = self::add_member($_REQUEST['sport'],$_REQUEST['season'],$athlete);
184                 $athlete_terms[] = $athlete['name'];
185             }
186             wp_set_post_terms($roster_id, $athlete_terms, 'sa_person', false );
187             if(isset($_POST['deleteObjects'])){
188                 self::delete_members($_POST['deleteObjects']);
189             }
190 
191             \SchoolAthletics\Admin\Notice::success('Roster has been successfully saved.');
192         }
193
