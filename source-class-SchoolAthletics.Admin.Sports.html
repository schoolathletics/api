227

229         if($options && !empty($sport)){
230

232                 $options['status'] = $status;
233             }
234

236                 $options['page_id'] = self::build_page($sport);
237             }
238 
239             if($options['roster']){
240                 if(!$options['roster_id']){
241                     $options['roster_id'] = self::build_roster_page($sport,$options['page_id']);
242                 }
243             }else{
244                 if($options['roster_id']){
245                     wp_delete_post( $options['roster_id'], true); 
246                     $options['roster_id'] = 0;
247                 }
248             }
249 
250             if($options['schedule']){
251                 if(!$options['schedule_id']){
252                     $options['schedule_id'] = self::build_schedule_page($sport,$options['page_id']);
253                 }
254             }else{
255                 if($options['schedule_id']){
256                     wp_delete_post( $options['schedule_id'], true); 
257                     $options['schedule_id'] = 0;
258                 }
259             }
260 
261             if($options['staff']){
262                 if(!$options['staff_id']){
263                     $options['staff_id'] = self::build_staff_page($sport,$options['page_id']);
264                 }
265             }else{
266                 if($options['staff_id']){
267                     wp_delete_post( $options['staff_id'], true); 
268                     $options['staff_id'] = 0;
269                 }
270             }

366                     $season = get_the_terms($page,'sa_season');
